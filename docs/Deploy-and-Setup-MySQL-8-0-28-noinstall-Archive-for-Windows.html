# Script Variables
$fileLocation = "c:\software\"
$mysqlPath = "C:\mysql\"
$zipFile = "mysql-8.0.28-winx64.zip"
$extractDirectory = "mysql-8.0.28-winx64"
$software = "MySQL Server"

# Check if the software is already installed.
$installed = (Test-Path $mysqlPath*)

	Write-Host "'$software' is already deployed."
} 
else 
{
# deploy software.
	Write-Host "Extract the '$software' files..."
	Expand-Archive -Path $fileLocation$zipFile -DestinationPath $fileLocation

	Move-Item -Path $fileLocation$extractDirectory\* -Destination $mysqlPath

([System.Environment]::GetEnvironmentVariable("Path", "Machine")).split(';') | foreach-object {

    {
        Write-Host "System Environment Path Variable already includes: " $_;
        Write-Host "System Environment Path Variable will not be updated.";
        $updatePathVariable = $false

    }
    else
    {
        $newPath = $newPath + $_ + ';'
    }
}

if ($updatePathVariable)
{
    $newPath = $newPath + $mysqlBinPath + ';'
    [System.Environment]::SetEnvironmentVariable("Path", $newPath, "Machine");

    # set datadir to the location of your data directory

{
    Write-Host "'$file' already exists."
} else
{
    New-Item -Path $path -Name "$file" -ItemType "file" -Value $contents

# Initialize MySQL.

{
    Write-Host "MySQL already initialized."
}
else
{
    mysqld --console --initialize-insecure

# Confirm initialization was successful.
    if (Test-Path $dataPath\*)
    {
        Write-Host "MySQL initialization successful."
